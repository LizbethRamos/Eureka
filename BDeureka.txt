drop database if exists eureka;
create database eureka;
USE eureka;

CREATE TABLE usuarios(
  id_usuario INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(128) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL,
  correo VARCHAR(128) NOT NULL,
  contrasenia VARCHAR(32) NOT NULL,
  tipo INT(1) NOT NULL
)ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

CREATE TABLE materias(
  id_materia INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  materia VARCHAR(64) CHARACTER SET utf8 COLLATE utf8_spanish_ci NOT NULL,
  imagen VARCHAR(256) NOT NULL,
  materia_activa INT DEFAULT 1
)ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

CREATE TABLE preguntas(
  id_pregunta INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  id_materia INT NOT NULL,
  nivel INT(1) NOT NULL,
  pregunta VARCHAR(512) NOT NULL,
  respuesta1 VARCHAR(256) NOT NULL,
  respuesta2 VARCHAR(256) NOT NULL,
  respuesta3 VARCHAR(256) NOT NULL, 
  respuesta_correcta VARCHAR(256) NOT NULL, 
  pregunta_activa INT DEFAULT 1,
  FOREIGN KEY(id_materia)REFERENCES materias(id_materia)
);


CREATE TABLE examenes(
   id_examen INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
   id_usuario INT NOT NULL,
   puntaje INT NOT NULL,
   inicio DATETIME NOT NULL,
   fin DATETIME NOT NULL,
   FOREIGN KEY(id_usuario)REFERENCES usuarios(id_usuario)
);

CREATE TABLE registros(
  id_examen INT NOT NULL,
  id_pregunta INT NOT NULL,
  respuesta VARCHAR(256) NOT NULL,
  FOREIGN KEY(id_examen)REFERENCES examenes(id_examen),
  FOREIGN KEY(id_pregunta)REFERENCES preguntas(id_pregunta),
  PRIMARY KEY (id_examen, id_pregunta)
);
